**Step 1: Check current Ember state on VPS**
```bash
ssh root@187.77.66.133
```

**Step 2: Deploy Ember code to VPS**

If Ember is not yet on the VPS:
```bash
# Option A: If promptheus-ember has a git remote
cd /home/ember 2>/dev/null || mkdir -p /home/ember
cd /home/ember
git clone <promptheus-ember-repo-url> .
npm ci --production

# Option B: If no git remote, scp from local machine

# Then on VPS:

```

If Ember already exists, update it:
```bash
cd /home/ember
git pull origin main  # or scp updated files
npm ci --production
```

**Step 3: Create Ember .env.local for YAPU**
```bash
cat > /home/ember/.env.local << 'EOF'
ANTHROPIC_API_KEY=sk-ant-REPLACE_WITH_REAL_KEY
AUTH_TOKEN=REPLACE_WITH_STRONG_TOKEN

WEBSITE_REPO_PATH=/home/yapu/yapu2
CUSTOMER_NAME=YAPU Solutions
CUSTOMER_TIER=2

PORT=3005
PREVIEW_PORT=3006

DOMAIN=ember.yapu.promptheus.cloud
PREVIEW_DOMAIN=preview.yapu.promptheus.cloud
WEBSITE_URL=https://yapu.promptheus.cloud
EOF
```

Replace `ANTHROPIC_API_KEY` with a real Anthropic API key and `AUTH_TOKEN` with a strong random token (e.g., `openssl rand -hex 32`).

**Step 4: Set up nginx for Ember**
```bash
# Create nginx config from template
cat > /etc/nginx/sites-available/ember-yapu << 'NGINX'
server {
    listen 80;
    server_name ember.yapu.promptheus.cloud;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    server_name ember.yapu.promptheus.cloud;

    ssl_certificate /etc/letsencrypt/live/ember.yapu.promptheus.cloud/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/ember.yapu.promptheus.cloud/privkey.pem;

    add_header X-Content-Type-Options "nosniff" always;
    add_header X-Frame-Options "DENY" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    proxy_buffering off;
    proxy_cache off;

    location / {
        proxy_pass http://127.0.0.1:3005;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 300s;
    }
}
NGINX

# Enable the site
ln -sf /etc/nginx/sites-available/ember-yapu /etc/nginx/sites-enabled/
nginx -t && systemctl reload nginx
```

**Step 5: SSL certificate (if not already exists)**
```bash
# Check if cert exists
ls /etc/letsencrypt/live/ember.yapu.promptheus.cloud/ 2>/dev/null

# If not, generate it (requires DNS A record already pointing to server):

# Then reload nginx:
systemctl reload nginx
```

